## 2.两数相加

- **原题链接：** [点击跳转](https://leetcode.cn/problems/add-two-numbers/?envType=study-plan-v2&envId=top-100-liked)

### 💡 思路历程
手动模拟，写了三层循环加最后的if判断，有些冗长，实际上，很多逻辑都是重复的，可以将他们都放入一个循环里。

当一条链表走完时，把它看作是一个值为 0 的无穷长链表。利用除法 / 获取进位，利用取模 % 获取当前位，省去 if-else 判断。
### 💻 C++ 冗长代码实现
```cpp
class Solution {
public:
    ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
       ListNode* dummy = new ListNode();
       ListNode* cur = dummy;
       int cod = 0;
        while(l1&&l2){
            int sum = l1->val+l2->val+cod;
            if(sum>=10){
                sum %= 10;
                cod = 1;
            }
            else{
                cod = 0;
            }
            ListNode* node = new ListNode(sum);
            cur->next = node;
            cur = cur->next;
            l1 = l1->next;
            l2 = l2->next;
        }
        
        while(l1){
            int sum = l1->val+cod;
            if(sum>=10){
                sum %= 10;
                cod = 1;
            }
            else{
                cod = 0;
            }
            ListNode* node = new ListNode(sum);
            cur->next = node;
            cur = cur->next;
            l1 = l1->next;
        }
        while(l2){
            int sum = l2->val+cod;
            if(sum>=10){
                sum %= 10;
                cod = 1;
            }
            else{
                cod = 0;
            }
            ListNode* node = new ListNode(sum);
            cur->next = node;
            cur = cur->next;
            l2 = l2->next;
        }
        if(!l1&&!l2&&cod == 1){
            ListNode* node = new ListNode(1);
            cur->next = node;
        }
        cur = dummy->next;
        delete(dummy);
        return cur；
    }
};
```

### 🥵💻 C++ 简洁代码实现
```cpp
class Solution {
public:
    ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
       ListNode* dummy = new ListNode();
       ListNode* cur = dummy;
       int carry = 0;
       while(l1||l2||carry){
            int val1 = l1?l1->val:0;
            int val2 = l2?l2->val:0;
            int sum = val1+val2+carry;
            carry = sum/10;
            ListNode* node = new ListNode(sum%10);
            cur->next = node;
            cur = cur->next;
            if(l1) l1 = l1->next;
            if(l2) l2 = l2->next;
       }
       ListNode* res = dummy->next;
        delete dummy; 
        return res;
    }
};
```
