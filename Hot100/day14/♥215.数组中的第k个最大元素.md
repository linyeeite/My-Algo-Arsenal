## 215.数组中的第k个最大元素
- **标签：** 堆/快排
- **原题链接：** [点击跳转](https://leetcode.cn/problems/kth-largest-element-in-an-array/description/?envType=study-plan-v2&envId=top-100-liked)

### 💡 思路历程

以此巩固快排的模板和堆的模板，

填坑法：双指针，一般取首点位数轴，也可随机取数然后在交换。

然后先从右边high往左边找第一个小于pivot的数，然后换到左边的位置，然后再从左边low往右找第一个大于pivot的数，同理。

细节在于找到等于的数时我们也要继续自增指标，否则会出现死循环，这一点是与hoare快排区别最大的点。

然后以上是划分函数，最后返回的是得以确定位置的点，然后再出个快排函数，得到返回的low，在对两边递归调用划分函数即可。注意退出条件。

### 💻 C++ 填坑法快排代码实现(标准法)
```cpp
class Solution {
public:
    int partition(vector<int>& nums,int low,int high) {
        int randomIndex = low + rand() % (high - low + 1); //取随机
        swap(nums[low], nums[randomIndex]);
        int pivot = nums[low];
        while(low<high){
            while(low<high&&nums[high]>=pivot) high--;
            nums[low] = nums[high];
            while(low<high&&nums[low]<=pivot) low++;
            nums[high] = nums[low];
        }
        nums[low] = pivot;
        return low;
    }

    void quicksort(vector<int>& nums,int low,int high){
        if(low>=high) return;
        int pivot = partition(nums,low,high);
        quicksort(nums,low,pivot-1);
        quicksort(nums,pivot+1,high);
    }
};
```

```cpp
class Solution {
public:
    int partition(vector<int>& nums,int low,int high) {

            while(low<high&&nums[low]<=pivot) low++;
            nums[high] = nums[low];
        }
        nums[low] = pivot;
        return low;
    }

    int quicksort(vector<int>& nums,int low,int high,int target){
        int pivot = partition(nums,low,high);
        if(pivot == target) return nums[pivot];
        if(pivot>target) return quicksort(nums,low,pivot-1,target);
        else return quicksort(nums,pivot+1,high,target);
    }

    int findKthLargest(vector<int>& nums, int k) {
        int n = nums.size();
        return quicksort(nums,0,n-1,n-k);
    }
};
```

### 💡 思路
  首先，同样是双指针，快排的灵魂是**“分治”**，而“归位”只是分治过程中的一个副产品。Hoare 抓住了分治的本质，舍弃了不必要的归位动作。体现在代码上，就是我们把=的情况也进行交换：
  
  “传统的填坑法虽然易于理解，但在处理大量重复元素时，由于必须使用带有等号的判断来避免死循环，会导致划分极其不均匀，时间复杂度退化。而 Hoare 划分通过 do-while 结构巧妙地避开了等号限制，强制相等元素进行交换，从而保证了分区的数学平衡性。这就是为什么 Hoare 版在机试中往往比填坑法快几个数量级的原因。”

  
### 💻 C++ hoare快排法代码实现
```cpp
int hoare_partition(vector<int>& nums, int l, int r) {
    // 取中间值作为枢轴，防止 O(n^2) 退化
    int pivot = nums[l + (r - l) / 2]; 
    int i = l - 1; 
    int j = r + 1;

    while (true) {
        // i 往右走，找大于等于 pivot 的数
        do { i++; } while (nums[i] < pivot);

        // j 往左走，找小于等于 pivot 的数
        do { j--; } while (nums[j] > pivot);

        if (i >= j) return j; // 关键：返回 j 作为分界点

        swap(nums[i], nums[j]); // 发现一对“站错队”的，交换
    }
}

void quick_sort(vector<int>& nums, int l, int r) {
    if (l >= r) return;

    // 拿到分界点
    int j = hoare_partition(nums, l, r);

    // 递归处理左右阵营
    quick_sort(nums, l, j);       // 注意：是 l 到 j
    quick_sort(nums, j + 1, r);   // 注意：是 j+1 到 r
}
```
### 💻 C++ hoare快排法代码实现（针对本题）

```cpp
class Solution {
public:
    int quickselect(vector<int> &nums, int l, int r, int k) {
        if (l == r)
            return nums[k];
        int partition = nums[l], i = l - 1, j = r + 1;
        while (i < j) {
            do i++; while (nums[i] < partition);
            do j--; while (nums[j] > partition);
            if (i < j)
                swap(nums[i], nums[j]);
        }
        if (k <= j)return quickselect(nums, l, j, k);
        else return quickselect(nums, j + 1, r, k);
    }

    int findKthLargest(vector<int> &nums, int k) {
        int n = nums.size();
        return quickselect(nums, 0, n - 1, n - k);
    }
};
```

### 💡 建堆思路
不调用优先队列的情况下，手写建堆过程。针对调整堆有两种写法，一种是迭代，一种是递归，王道书上的是迭代，用for循环以此向下调整，同时下标是从1开始的，0作为哨兵结点保存原值。个人认为递归更易理解，创一个变量记录原下标，从0开始即可。不过迭代法很优美，向下传递的美感。在此写下迭代代码，从0起，递归就不写了。
### 💻 C++ 堆迭代代码实现
```cpp
class Solution {
public:
    // 迭代下沉函数
    void heapify(vector<int>& nums, int idx, int heapsize) {
        while (true) {
            int largest = idx;
            int l = 2 * idx + 1;
            int r = 2 * idx + 2;
            if (l < heapsize && nums[l] > nums[largest])
                largest = l;
            if (r < heapsize && nums[r] > nums[largest])
                largest = r;
            if (largest == idx)
                break;
            swap(nums[idx], nums[largest]);
            idx = largest;
        }
    }

    // 构建大顶堆
    void buildMaxHeap(vector<int>& nums, int n) {
        for (int i = n / 2 - 1; i >= 0; i--) {
            heapify(nums, i, n);
        }
    }

    int findKthLargest(vector<int>& nums, int k) {
        int n = nums.size();
        buildMaxHeap(nums, n);

        int heapsize = n;
        for (int i = 0; i < k - 1; i++) {
            swap(nums[0], nums[heapsize - 1]); // 把堆顶移到末尾
            heapsize--;                         // 堆大小减1
            heapify(nums, 0, heapsize);         // 调整堆
        }
        return nums[0]; // 堆顶就是第k大
    }
};
```
当然，也可以不用手建堆，直接用stl，且用小根堆，将空间利用到极致。
### 💻 C++ stl代码实现
```cpp
class Solution {
public:
    int findKthLargest(vector<int>& nums, int k) {
        // 定义小顶堆 (Min-Heap)
        // 注意参数：vector 是底层容器，greater<int> 决定了它是小顶堆
        priority_queue<int, vector<int>, greater<int>> pq;

        for (int num : nums) {
            if (pq.size() < k) {
                pq.push(num);
            } else if (num > pq.top()) {
                pq.pop();
                pq.push(num);
            }
        }
        
        return pq.top();
    }
};
```
