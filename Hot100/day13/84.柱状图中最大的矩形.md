## 84.柱状图中最大的矩形
- **标签：** 单调栈
- **原题链接：** [点击跳转](https://leetcode.cn/problems/largest-rectangle-in-histogram/?envType=study-plan-v2&envId=top-100-liked)

### 💡 思路历程
两种做法，一种单调栈，注意好width的大小别弄错了。一种用数组记录左边右边第一个小于本点高度的对应下标，然后求即可，相对像像人类思维。

在这里再阐述下第二次做过遇到的坑：

1.width宽度忘记减一，因为我们取出对应下标后，pop了，所以此时st.top()是左边界，而i是右边界，所以要减一。

2.因为我们的单调栈是单调递减栈（从栈顶往栈底看），所以不对原高度数组做处理的话，最后栈可能会留下标导致面积未能计算完全，此时我们可以通过加哨兵结点，前后都加个0，此时即可弹出所有有效高度结点。
### 💻 C++ 单调栈代码实现
```cpp
class Solution {
public:
    int largestRectangleArea(vector<int>& heights) {
        stack<int> st;
        int result = 0;
        heights.insert(heights.begin(),0);
        heights.push_back(0);
        st.push(0);
        for(int i = 1;i<heights.size();i++){
            while(!st.empty()&&heights[i]<heights[st.top()]){
                int mid = st.top();
                st.pop();
                if(!st.empty()){
                    int left = st.top();
                    int right = i;
                    int count = (right - left - 1)*heights[mid];
                    result = max(result,count);
                }
                
            }
            st.push(i);
        }
        
        return result;
    }
};
```

### 💻 C++ 双指针代码实现
```cpp
class Solution {
public:
    int largestRectangleArea(vector<int>& heights) {
        int n = heights.size();
        int result = 0;
        vector<int> minL(n,0);
        vector<int> minR(n,0);
        minL[0] = -1;
        for(int i = 1;i<n;i++){
            int t = i-1;
            while(t>=0 && heights[t]>=heights[i]) t = minL[t];
            minL[i] = t;
        }
        minR[n-1] = n;
        for(int i = n-2;i>=0;i--){
            int t = i+1;
            while(t<n && heights[t]>=heights[i]) t = minR[t];
            minR[i] = t;
        }
        for(int i = 0;i<n;i++){
            int lenth = minR[i] - minL[i]-1;
            int count = heights[i]*lenth;
            result = max(result,count);
        }
        return result;
    }
};
```
